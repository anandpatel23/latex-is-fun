	% Insert algorithm
	\begin{algorithm}
		\caption{compute sum of integers in array}
		\label{array-sum}
		\begin{algorithmic}[1]
			\Procedure{ArraySum}{$A$}
			\State $sum = 0$
			\For {each integer $i$ in $A$}
			\State $sum = sum + i$
			\EndFor
			\State Return $sum$
			\EndProcedure
		\end{algorithmic}
	\end{algorithm}
