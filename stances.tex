	\begin{algorithm}
		\caption{buy when not invested and stock price drops, sell when price is greater than 0.2 percent higher than bought}
		\label{macd}
		\begin{algorithmic}[1]
			\If {$stance$ is 'none'}
				\If {$price$ is less than $previousPrice$}
					\State buy triggered
					\State $stance$ = 'holding'
					\If{$tradeCount$ is 0}
						\State $startingPrice$ = $priceBought$
					\EndIf
					\State $tradeCount$ += 1
				\EndIf
			\ElsIf{$stance$ is 'holding}
				\If{price is greater than ($priceBought$ * 0.002 + $priceBought$)}
					\State sell triggered
					\State $priceSold$ = $price$
					\State $stance$ = 'none'
					\State $tradeProfit$ = $priceSold$ - $priceBought$
					\State $totalProfit$ += $tradeProfit$
					\State $tradeCount$ += 1
				\EndIf
			\EndIf
			\State $previousPrice$ = $price$			
		\end{algorithmic}
	\end{algorithm}
